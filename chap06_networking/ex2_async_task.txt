Open HelloWorldDemo in Eclipse
Create an async task that logs to a 100,000 in the background and
  displays a Toast when it's done counting.

## Snippets

// In Activity
@Override
protected void onCreate(Bundle savedInstanceState) {
   // ...
   new MyAsyncTask().execute();
}

public void doneCounting() {
  Toast.makeText(this, "Done Counting to 100000", Toast.LENGTH_SHORT).show();
}

private class MyAsyncTask extends AsyncTask<Void, Void, Void> {
    public Void doInBackground(Void... params) {
      for (long i=0; i < 100000; i++) {
        System.out.println(i);
      }
      return null;
    }

    protected void onPostExecute(Void result) {
      doneCounting();
    }
}
