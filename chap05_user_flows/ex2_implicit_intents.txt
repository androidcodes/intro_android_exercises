Open HelloWorldDemo in Eclipse
In the first activity, setup a TextEdit to specify a url address to visit.
Create a "Visit" button that when clicked opens up the specified url
  in a browser using an ACTION_VIEW implicit intent


## Snippets

// In Java Activity
public void visitUrlAddress(View v) {
  Uri url = getUriToVisit();
  if (url != null) {
    Intent i = new Intent(Intent.ACTION_VIEW);
    i.setData(url);
    startActivity(i);
  }

}

public Uri getUriToVisit() {
  String urlAddress =  ((TextView) findViewById(R.id.txtUrlAddress)).getText().toString();
  if (urlAddress != null) {
    if (!urlAddress.startsWith("http://"))
       urlAddress = "http://" + urlAddress;
    return Uri.parse(urlAddress);
  } else {
    return null;
  }
}
