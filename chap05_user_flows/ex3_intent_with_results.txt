Open HelloWorldDemo in Eclipse
Create at least one contact within the emulator or device
Create or open a new activity
When a button is clicked, have another activity started using startActivityforResult
In the second activity, have the user enter text and hit a button and return the text as a result
In the first activity, display the text result as a toast message

## Snippets

// In Activity
final static int GET_RESULT_TEXT = 0;

public void enterText(View v) {
  startActivityForResult(
        new Intent(IntentWithResultActivity.this, SimpleReturnResultActivity.class),
          GET_RESULT_TEXT);
}

// Handle the result once the activity returns a result, display contact
protected void onActivityResult(int requestCode, int resultCode, Intent data) {
  if (requestCode == 0) {
    if (resultCode == RESULT_OK) {
      TextView tvResult = (TextView)findViewById(R.id.txtDisplayResult);
      tvResult.setText(data.getStringExtra("result"));
      Toast.makeText(this, data.getStringExtra("result"), Toast.LENGTH_SHORT).show();
    }
  }
}

// Returning Activity
public void sendResult(View v) {
  String result = ((EditText) findViewById(R.id.txtRandomResultText)).getText().toString();
  Intent i = new Intent();
  i.putExtra("result", result);
  setResult(RESULT_OK, i);
  finish();
}
